\def\clap#1{\hbox to 0pt{\hss#1\hss}}
\def\mathllap{\mathpalette\mathllapinternal}
\def\mathrlap{\mathpalette\mathrlapinternal}
\def\mathclap{\mathpalette\mathclapinternal}
\def\mathllapinternal#1#2{%
\llap{$\mathsurround=0pt#1{#2}$}}
\def\mathrlapinternal#1#2{%
\rlap{$\mathsurround=0pt#1{#2}$}}
\def\mathclapinternal#1#2{%
\clap{$\mathsurround=0pt#1{#2}$}}

\NewDocumentCommand\Decomposition{}{%
  \quad\overset{\mathclap{\text{Dekomp.}}}{\Rightarrow}\quad%
}
\NewDocumentCommand\Eliminate{}{%
  \quad\overset{\mathclap{\text{Elim.}}}{\Rightarrow}\quad%
}
\NewDocumentCommand\Substitute{}{%
  \quad\overset{\mathclap{\text{Subst.}}}{\Rightarrow}\quad%
}
\NewDocumentCommand\Swap{}{%
  \quad\overset{\mathclap{\text{Vert.}}}{\Rightarrow}\quad%
}
