\begin{frame}{Aufgabe 1a}
Für A soll gelten:
  \begin{gather*}
    A t s u \Rightarrow_{β}^{*} s \\
    \uncover<2->{
      A = λ x y z. y \\
    }
    \uncover<3->{
      (λ x y z. y) t s u 
    }
    \uncover<4->{
      \Rightarrow_{β}(λ y z. y) s u 
    }
    \uncover<5->{
      \Rightarrow_{β} (λ z. s) u 
    }
    \uncover<6->{
      \Rightarrow_{β} s 
    }
  \end{gather*}
\end{frame}
